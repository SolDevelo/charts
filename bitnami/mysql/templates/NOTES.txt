  echo Username: root
  MYSQL_ROOT_PASSWORD=$(kubectl get secret --namespace {{ include "common.names.namespace" . }} {{ template "mysql.secretName" . }} -o jsonpath="{.data.mysql-root-password}" | base64 -d)

To connect to your database, create a client pod:

    kubectl run {{ include "common.names.fullname" . }}-client --rm --tty -i --restart='Never' --image  {{ template "mysql.image" . }} --namespace {{ include "common.names.namespace" . }} --env MYSQL_ROOT_PASSWORD=$MYSQL_ROOT_PASSWORD --command -- bash

{{- if .Values.tls.enabled }}

Copy your CA TLS certificates to your client pod:

    kubectl get secret --namespace {{ include "common.names.namespace" . }} {{ include "mysql.tls.secretName" . }} -o json | jq -r '.data["{{ ternary "ca.crt" .Values.tls.certCAFilename .Values.tls.autoGenerated.enabled }}"]' | base64 --decode > /tmp/ca.crt
    kubectl cp --namespace {{ include "common.names.namespace" . }} /tmp/ca.crt {{ include "common.names.fullname" . }}-client:/tmp/ca.crt

{{- end }}
