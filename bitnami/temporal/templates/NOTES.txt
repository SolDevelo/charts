To access Temporal Frontend gRPC API from outside the cluster execute the following commands:

1. Get the Temporal Frontend gRPC API URL by running these commands:

{{- if contains "NodePort" .Values.frontend.service.type }}

    echo "Host: $(kubectl get nodes --namespace {{ include "common.names.namespace" . }} -o jsonpath="{.items[0].status.addresses[0].address}")"
    echo "Port: $(kubectl get --namespace {{ include "common.names.namespace" . }} -o jsonpath="{.spec.ports[?(@.name=='grpc-rpc')].nodePort}" services {{ include "common.names.fullname" . }})"

{{- else if contains "LoadBalancer" .Values.frontend.service.type }}

  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        You can watch its status by running 'kubectl get --namespace {{ include "common.names.namespace" . }} svc -w {{ include "temporal.frontend.fullname" . }}'

    echo "Host: $(kubectl get svc --namespace {{ include "common.names.namespace" . }} {{ include "temporal.frontend.fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')"
    echo "Port: $(kubectl get --namespace {{ include "common.names.namespace" . }} -o jsonpath="{.spec.ports[?(@.name=='grpc-rpc')].port}" services {{ include "temporal.frontend.fullname" . }})"

{{- else if contains "ClusterIP" .Values.frontend.service.type }}

    export SERVICE_PORT=$(kubectl get --namespace {{ include "common.names.namespace" . }} -o jsonpath="{.spec.ports[?(@.name=='grpc-rpc')].port}" services {{ include "temporal.frontend.fullname" . }})
    kubectl port-forward --namespace {{ include "common.names.namespace" . }} svc/{{ include "temporal.frontend.fullname" . }} ${SERVICE_PORT}:${SERVICE_PORT} &
    echo "Host: 127.0.0.1"
    echo "Port: ${SERVICE_PORT}"

{{- end }}
